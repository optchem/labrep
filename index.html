random.seed(12345)

v1 = 100
v2 = 300
q01 = 0.2
q02 = 0.25
q12 = 0.06
q21 = 0.08
q10 = 0.19
q20 = 0.26
c01 = 25
c02 = 18

tspan = (0., 6000.)
l, h = tspan

c1_measured = [0.5, 1.5, 2.24647738e+01, 2.28559236e+01, 2.54612099e+01,
       2.55141833e+01, 2.55567227e+01, 2.55903438e+01, 2.56164883e+01,
       2.56364748e+01, 2.56514748e+01, 2.56625060e+01, 2.56628613e+01]

t_measured = np.linspace(l,h,len(c1_measured))
t_measured += np.random.randint(1, size=t_measured.size)

u0 = [0, 0]

def f(t, u, _c01):
    
    c01 = _c01
    # print(c01)
    
    c1, c2 = u
    rc1 = (q01*c01+q21*c2-q12*c1-q10*c1)/v1
    rc2 = (q02*c02+q12*c1-q21*c2-q20*c2)/v2
    return [rc1, rc2]

def f_base(t, u):

    rc2 = (q02*c02+q12*c1-q21*c2-q20*c2)/v2
    return [rc1, rc2]
